# T32 — [Phase 3] — Art score voting UI

## Overview
This task gives players a tactile 5-star rating interface for appreciating other players' sand castles. The star widget appears at the bottom of the screen only when viewing another player's plot, is disabled when the daily 10-vote limit is reached, and provides satisfying feedback through gold star fills and a Korean toast notification. The system is deeply integrated with the realtime score API (T31) and triggers leaderboard updates (T33) after each vote to keep rankings fresh.

## Scope
Modify `src/ui/hud.ts` to render a star rating widget at the bottom-center of the screen when `isViewingOtherPlot === true`. The widget consists of 5 star shapes (30×30px each, 8px gap), drawn unfilled gray (`#cccccc`) by default and filled gold (`#ffc107`) for selected and previously-voted stars. Tapping a star selects that score (1–5). A "확인" submit button (80×32px, blue) appears to the right of the stars after a selection. Selecting a new star before submitting simply changes the local selection. On submit: call `submitScore(plotId, currentUserId, selectedScore)`, then call `updateLeaderboard(plotOwnerUserId, newAvgScore)` (from T33), then display a Korean toast message `"감사합니다! ⭐"` for 2 seconds (300ms fade in, 1400ms hold, 300ms fade out). Hide the widget entirely (`isViewingOtherPlot === false`) when on own plot. Disable the widget (stars rendered with 40% opacity, button grayed out, tap ignored) when `getTodayVoteCount(currentUserId) >= 10`. Load the current vote count once when entering a plot view, not every frame.

### Entering a plot view

```typescript
export async function enterPlotView(
  plotId: string,
  plotOwnerId: string,
  currentUserId: string,
  isOwnPlot: boolean
): Promise<void> {
  if (isOwnPlot) {
    starWidget.visible = false;
    return;
  }

  starWidget.visible = true;
  starWidget.plotId = plotId;
  starWidget.plotOwnerId = plotOwnerId;
  starWidget.selectedScore = null;
  starWidget.isSubmitting = false;

  // Load vote count (once, not every frame)
  const [voteCount, plotScore] = await Promise.all([
    getTodayVoteCount(currentUserId),
    getPlotScore(plotId),
  ]);

  starWidget.todayVoteCount = voteCount;
  starWidget.disabled = voteCount >= 10;
}

export function leavePlotView(): void {
  starWidget.visible = false;
  starWidget.plotId = null;
  starWidget.plotOwnerId = null;
}
```

### Star drawing helper

```typescript
const STAR_SIZE = 30;
const STAR_GAP = 8;
const STARS_TOTAL_WIDTH = 5 * STAR_SIZE + 4 * STAR_GAP; // 190px

function drawStar(
  ctx: CanvasRenderingContext2D,
  cx: number,
  cy: number,
  size: number,
  filled: boolean,
  disabled: boolean
): void {
  const outerR = size / 2;
  const innerR = outerR * 0.4;
  const points = 5;

  ctx.save();
  ctx.globalAlpha = disabled ? 0.4 : 1.0;
  ctx.fillStyle = filled ? '#ffc107' : '#cccccc';
  ctx.strokeStyle = filled ? '#e6a800' : '#aaaaaa';
  ctx.lineWidth = 1;

  ctx.beginPath();
  for (let i = 0; i < points * 2; i++) {
    const angle = (i * Math.PI) / points - Math.PI / 2;
    const r = i % 2 === 0 ? outerR : innerR;
    const x = cx + Math.cos(angle) * r;
    const y = cy + Math.sin(angle) * r;
    if (i === 0) ctx.moveTo(x, y);
    else ctx.lineTo(x, y);
  }
  ctx.closePath();
  ctx.fill();
  ctx.stroke();
  ctx.restore();
}
```

### renderStarWidget

```typescript
const WIDGET_BOTTOM_MARGIN = 80; // px from canvas bottom

export function renderStarWidget(
  ctx: CanvasRenderingContext2D,
  canvasWidth: number,
  canvasHeight: number
): void {
  if (!starWidget.visible) return;

  const widgetY = canvasHeight - WIDGET_BOTTOM_MARGIN;
  const starsStartX = (canvasWidth - STARS_TOTAL_WIDTH) / 2;

  // Background pill
  ctx.fillStyle = 'rgba(0, 0, 0, 0.6)';
  ctx.beginPath();
  ctx.roundRect(starsStartX - 16, widgetY - 20, STARS_TOTAL_WIDTH + 120, 60, 12);
  ctx.fill();

  // Draw 5 stars
  for (let i = 0; i < 5; i++) {
    const starScore = (i + 1) as StarScore;
    const cx = starsStartX + i * (STAR_SIZE + STAR_GAP) + STAR_SIZE / 2;
    const cy = widgetY + 10;
    const filled =
      starWidget.selectedScore !== null
        ? starScore <= starWidget.selectedScore
        : starWidget.existingScore !== null
        ? starScore <= starWidget.existingScore
        : false;

    drawStar(ctx, cx, cy, STAR_SIZE, filled, starWidget.disabled);
  }

  // Submit button (only show after a star is selected)
  if (starWidget.selectedScore !== null && !starWidget.isSubmitting) {
    const btnX = starsStartX + STARS_TOTAL_WIDTH + 16;
    const btnY = widgetY - 6;
    const btnW = 64;
    const btnH = 32;

    ctx.fillStyle = starWidget.disabled ? '#888888' : '#1976d2';
    ctx.beginPath();
    ctx.roundRect(btnX, btnY, btnW, btnH, 6);
    ctx.fill();

    ctx.fillStyle = '#ffffff';
    ctx.font = 'bold 14px "Apple SD Gothic Neo", sans-serif';
    ctx.textAlign = 'center';
    ctx.fillText('확인', btnX + btnW / 2, btnY + 21);
  }

  // Disabled notice
  if (starWidget.disabled) {
    ctx.fillStyle = 'rgba(255,255,255,0.6)';
    ctx.font = '11px "Apple SD Gothic Neo", sans-serif';
    ctx.textAlign = 'center';
    ctx.fillText('오늘 투표 한도에 도달했습니다', canvasWidth / 2, widgetY + 42);
  }
}
```

### Star tap hit-testing

Add to the canvas `pointerdown` handler:

```typescript
function handleStarWidgetTap(
  canvasX: number,
  canvasY: number,
  canvasWidth: number,
  canvasHeight: number,
  currentUserId: string
): void {
  if (!starWidget.visible || starWidget.disabled || starWidget.isSubmitting) return;

  const widgetY = canvasHeight - WIDGET_BOTTOM_MARGIN;
  const starsStartX = (canvasWidth - STARS_TOTAL_WIDTH) / 2;

  // Check each star
  for (let i = 0; i < 5; i++) {
    const starScore = (i + 1) as StarScore;
    const cx = starsStartX + i * (STAR_SIZE + STAR_GAP) + STAR_SIZE / 2;
    const cy = widgetY + 10;
    const dx = canvasX - cx;
    const dy = canvasY - cy;
    if (Math.sqrt(dx * dx + dy * dy) <= STAR_SIZE / 2 + 4) {
      starWidget.selectedScore = starScore;
      return;
    }
  }

  // Check submit button
  if (starWidget.selectedScore !== null) {
    const btnX = starsStartX + STARS_TOTAL_WIDTH + 16;
    const btnY = widgetY - 6;
    if (canvasX >= btnX && canvasX <= btnX + 64 && canvasY >= btnY && canvasY <= btnY + 32) {
      handleSubmitVote(currentUserId);
    }
  }
}
```

### handleSubmitVote

```typescript
async function handleSubmitVote(currentUserId: string): Promise<void> {
  if (!starWidget.plotId || !starWidget.selectedScore || starWidget.isSubmitting) return;

  starWidget.isSubmitting = true;
  const score = starWidget.selectedScore;
  const plotId = starWidget.plotId;
  const plotOwnerId = starWidget.plotOwnerId!;

  const result = await submitScore(plotId, currentUserId, score);

  if (result.success) {
    const plotScore = await getPlotScore(plotId);
    await updateLeaderboard(plotOwnerId, plotScore.avg);

    starWidget.existingScore = score;
    starWidget.selectedScore = null;
    starWidget.todayVoteCount++;
    starWidget.disabled = starWidget.todayVoteCount >= 10;

    showToast('감사합니다! ⭐');
  } else {
    showToast('투표 실패: ' + result.error);
  }

  starWidget.isSubmitting = false;
}
```

### Toast render and animation

```typescript
function showToast(message: string): void {
  toast.message = message;
  toast.phase = 'fadein';
  toast.startTime = Date.now();
  toast.alpha = 0;
}

export function renderVoteToast(
  ctx: CanvasRenderingContext2D,
  canvasWidth: number,
  canvasHeight: number
): void {
  if (toast.phase === 'idle') return;

  const elapsed = Date.now() - toast.startTime;
  const FADE_IN = 300;
  const HOLD = 1400;
  const FADE_OUT = 300;

  if (elapsed < FADE_IN) {
    toast.alpha = elapsed / FADE_IN;
  } else if (elapsed < FADE_IN + HOLD) {
    toast.alpha = 1.0;
  } else if (elapsed < FADE_IN + HOLD + FADE_OUT) {
    toast.alpha = 1 - (elapsed - FADE_IN - HOLD) / FADE_OUT;
  } else {
    toast.phase = 'idle';
    return;
  }

  const toastWidth = 200;
  const toastX = (canvasWidth - toastWidth) / 2;
  const toastY = canvasHeight - 160;

  ctx.save();
  ctx.globalAlpha = toast.alpha;
  ctx.fillStyle = 'rgba(0,0,0,0.75)';
  ctx.beginPath();
  ctx.roundRect(toastX, toastY, toastWidth, 36, 8);
  ctx.fill();

  ctx.fillStyle = '#ffffff';
  ctx.font = 'bold 15px "Apple SD Gothic Neo", sans-serif';
  ctx.textAlign = 'center';
  ctx.fillText(toast.message, canvasWidth / 2, toastY + 23);
  ctx.restore();
}
```

## Key Constraints
- The widget must be completely hidden (`visible = false`) on the player's own plot — never show a self-vote option.
- `getTodayVoteCount` is called once on `enterPlotView`, not every frame — avoids spamming Supabase on each render.
- `isSubmitting` blocks double-submit: the button tap is ignored while a network call is in flight.
- `updateLeaderboard` is called after a successful vote — import from `src/ait/leaderboard.ts` (T33).
- Stars use a hit radius of `STAR_SIZE/2 + 4px` (34px) for comfortable touch targets on mobile.
- The toast fades in/out using `ctx.globalAlpha` — ensure `ctx.restore()` is always called after rendering.

## Acceptance Criteria
- [ ] Star widget renders at the bottom of screen only when `isViewingOtherPlot === true`.
- [ ] Stars 1–N are filled gold when score N is selected; stars above N are gray.
- [ ] Tapping a star updates the selection; tapping "확인" submits via `submitScore`.
- [ ] After submit, a "감사합니다! ⭐" toast appears, holds for ~1.4s, and fades out.
- [ ] Widget is rendered at 40% opacity and taps are ignored when `todayVoteCount >= 10`.
- [ ] The widget is completely absent when viewing own plot.
- [ ] `updateLeaderboard` is called with the plot owner's userId and the new average score after each successful vote.
- [ ] `isSubmitting` prevents duplicate submits if the user taps the button multiple times rapidly.
